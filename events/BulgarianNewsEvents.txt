news_event = {
	id = bulgarian_news.3
	hidden = yes  #timeout
    timeout_days = 1
	is_triggered_only = yes
	fire_only_once = yes
	
	trigger = {
		BUL = {
			NOT = {
				has_completed_focus = BUL_revise_our_regime
			}
		}
		#date > 1943.8.27
		#date < 1943.8.30
		country_exists = BUL
	}
	
	immediate = {
		if = {
			limit = {
				check_variable = { global.event_random < 50 }
			}
			news_event = { id = bulgarian_news.4 }
		}
		else = {
			news_event = { id = bulgarian_news.5 }
		}
	}
}
