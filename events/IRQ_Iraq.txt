#Ghazi I Death

country_event = {
	id = iraq.4
	title = iraq.4.t 
	desc = iraq.4.d
	picture = GFX_report_event_state_funeral_1_gfxaddon
	
	fire_only_once = yes
	is_triggered_only = yes
	trigger = {
		tag = IRQ
	}
	
	option = { # Abd al-Ilah
		name = iraq.4.a
		IRQ_ghazi_i = { kill_character = yes }
		if = {
			limit = { has_government = monarchism }
			add_popularity = {
					ideology = monarchism
					popularity = 0.05
			}
			add_advisor_role = {
				character = IRQ_abd_al-ilah
				advisor = {
					slot = head_of_state
					idea_token = IRQ_abd_al-ilah_head_of_state		
					traits = { 
						monarchist
						naive_appeaser
					}
				}
				activate = yes
			}
		}
	}
}

#### Coup and ENG declares war AI v AI###
country_event = {
	id = iraq.8
	hidden = yes 
	
	fire_only_once = yes
	
	trigger = {
		is_ai = yes
		is_subject = no
	}

	is_triggered_only = yes
	
	option = {
		name = ok
		ai_chance = { factor = 100 }
		set_politics = {
			ruling_party = fascism
		}
		add_popularity = {
			ideology = fascism
			popularity = 0.15
		}
		if = {
			limit = {
				ENG = {
					is_ai = no
				}
			}
			country_event = { id = iraq.10 days = 35 }
		}
		else = {
			ENG = {
				country_event = { id = iraq.9 days = 31 }
			}
		}
	}
}
