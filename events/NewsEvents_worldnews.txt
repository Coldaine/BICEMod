# Paris Universal Exhibition
news_event = {
	id = worldnews.5
	title = worldnews.5.t
	desc = {
		trigger = {
			SOV = {
				not = { has_country_flag = paris_prep }
			}
		}
		text = worldnews.5.d
	}
	desc = {
		trigger = {
			SOV = {
				has_country_flag = paris_prep
			}
		}
		text = worldnews.5.p
			
	}
	picture = GFX_news_event_paris_exhibition_1937
	is_triggered_only = yes
	major = yes
	trigger = {
		FRA = { has_war = no }
	}
	immediate = {
		set_global_flag = paris_fair
	}
	option = { 
		trigger = {
			OR = {
				SOV = {
					not = { has_country_flag = paris_prep }
				}
				NOT = { tag = SOV }
			}
		}
		name = worldnews.5.a
		if = {
			limit = {
				SOV = { has_country_flag = paris_prep }
				capital_scope = {
					is_on_continent = europe
				}
			}
			effect_tooltip = {
				add_timed_idea = {
					idea = soviet_fair
					days = 184
				}
			}
		}		
	}
	option = { 
		trigger = {
			SOV = {
				has_country_flag = paris_prep
			}
			tag = SOV
		}
		every_other_country = { 
			add_opinion_modifier = {
				target = SOV
				modifier = world_fair
			}
		}
		every_other_country = { 
			limit = {
				capital_scope = {
					is_on_continent = europe
				}
			}
			add_timed_idea = {
				idea = soviet_fair
				days = 184
			}
		}		
		name = worldnews.5.b 
	}	
	
}
