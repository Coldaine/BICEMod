		ai_chance = {
			factor = 50
		}

			limit = {
				OR = {
					check_variable = { global.event_random < 60 }
					AND = {
						OR = {
							has_idea_with_trait = tough_negotiator
							has_idea_with_trait = smooth_talking_charmer
							has_idea_with_trait = tough_guy

					}
				}

		else = {
			custom_effect_tooltip = foreign_minister_buy_good

		}		
	}

		ai_chance = {
			factor = 50
		}
